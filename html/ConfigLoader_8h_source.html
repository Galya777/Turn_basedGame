126 void parseCommanderLine(const std::string& line, bool isLiving) {
127 std::istringstream ss(line);
128 std::string name, armorName, unitListStr;
129 int hp = 0, mana = 0, gold = 0;
130
131 std::getline(ss, name, ',');
132 ss >> hp; ss.ignore();
133 ss >> mana; ss.ignore();
134 ss >> gold; ss.ignore();
135 std::getline(ss, armorName, ',');
136 std::getline(ss, unitListStr);
137
138 if (commanderNameExists(name)) {
139 std::cerr << "[!] Comander already exists: " << name << "\n";
140 return;
141 }
142
143 Armor* armor = createArmor(armorName);
144 Comander* commander = nullptr;
145
146 // Специални командири
147 if (name == "Lich") commander = new Lich(name);
148 else if (name == "DeathLord") commander = new DreadLord(name);
149 else if (name == "DeadKnight") commander = new DeathKnight(name);
150 else if (name == "UndeadHunter") commander = new UndeadHunter(name);
151 else if (name == "Bladedancer") commander = new DancerWithBlades(name);
152 else if (name == "Paladin") commander = new Paladin(name);
153 else commander = new Comander(name, hp, mana, armor, gold, 0);
154
155 // Раздели units
156 std::vector<std::string> unitTypes;
157 std::istringstream unitSS(unitListStr);
158 std::string unit;
159 while (std::getline(unitSS, unit, ';')) {
160 unitTypes.push_back(unit);
161 }
162
163 assignUnitsToCommander(commander, unitTypes);
164
165 if (isLiving) livingCommanders.push_back(commander);
166 else undeadCommanders.push_back(commander);
167 }

169 void loadCommandersFromFile(const std::string& filename, bool isLiving) {

179 parseCommanderLine(line, isLiving);
